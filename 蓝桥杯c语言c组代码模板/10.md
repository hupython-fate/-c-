### 补全10. 单向链表基础模板（补上链表考点，最后一个模板）
```c
#include <stdio.h>
#include <stdlib.h>
// 链表的节点结构
typedef struct Node {
    int data;          // 节点存的数据
    struct Node *next; // 指向下一个节点的指针
} Node;

// 创建一个新的节点
Node* create_node(int x) {
    Node *node = (Node*)malloc(sizeof(Node));
    node->data = x;
    node->next = NULL;
    return node;
}

// 尾插法：把x插到链表的最后
void insert_tail(Node *head, int x) {
    Node *p = head;
    // 找到最后一个节点
    while (p->next != NULL) {
        p = p->next;
    }
    p->next = create_node(x);
}

// 删除值为x的节点
void delete_node(Node *head, int x) {
    Node *p = head;
    while (p->next != NULL) {
        if (p->next->data == x) {
            // 把当前节点的next，指向下下个节点，跳过要删的节点
            Node *tmp = p->next;
            p->next = p->next->next;
            free(tmp); // 释放内存
            return;
        }
        p = p->next;
    }
}

// 遍历链表，输出所有元素
void traverse(Node *head) {
    Node *p = head->next;
    while (p != NULL) {
        printf("%d ", p->data);
        p = p->next;
    }
    printf("\n");
}

int main() {
    // 示例：链表的基础操作
    Node *head = create_node(0); // 头节点，不存数据，方便操作
    insert_tail(head, 1);
    insert_tail(head, 2);
    insert_tail(head, 3);
    printf("插入后的链表：");
    traverse(head); // 输出 1 2 3

    delete_node(head, 2);
    printf("删除2后的链表：");
    traverse(head); // 输出 1 3
    return 0;
}
```
> 背诵点：链表的核心就是**节点+指针**，每个节点存自己的数据，和下一个节点的地址，C组的链表都是基础的增删查，背会这个，直接套就行，不会考复杂的环、反转这些。
