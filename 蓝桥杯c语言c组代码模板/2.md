### 4. 字符串基础：读取、遍历、计数、反转模板
```c
#include <stdio.h>
#include <string.h> // 字符串处理头文件

int main() {
    char str[1005];
    scanf("%s", str);          // 读入字符串，自动忽略空格，读到换行结束
    int len = strlen(str);     // 快速拿到字符串长度，直接用标准库就行

    // 1. 遍历字符串的每个字符（所有字符串题的基础）
    for (int i = 0; i < len; i++) {
        // 可以在这里对每个字符做处理：比如转大小写、判断、计数
        printf("%c ", str[i]);
    }
    printf("\n");

    // 2. 统计某个字符出现的次数（C组高频题）
    char target;
    scanf(" %c", &target);    // 前面加空格，跳过之前的换行符，避免读错
    int cnt = 0;
    for (int i = 0; i < len; i++) {
        if (str[i] == target) cnt++;
    }

    // 3. 反转字符串（常考操作）
    for (int i = 0; i < len/2; i++) {
        // 交换左右对称位置的字符
        char tmp = str[i];
        str[i] = str[len - 1 - i];
        str[len - 1 - i] = tmp;
    }

    printf("字符%c出现次数：%d，反转后字符串：%s\n", target, cnt, str);
    return 0;
}
```
> 背诵点：C组字符串题核心就是**字符数组遍历**，所有的计数、替换、反转都是在遍历里改，不用搞复杂的函数，strlen直接用就行，省时间。

### 5. 暴力枚举基础模板【官方大纲明确考点：枚举[1-3]】
```c
#include <stdio.h>
int main() {
    // 示例：找1~100中，两个数和为100的所有组合，其他题改范围和条件就行
    int target = 100;

    // 👇 核心：双层for穷举所有可能的情况，C组数据小，暴力100%能过
    for (int i = 1; i <= 100; i++) {       // 枚举第一个变量
        for (int j = i + 1; j <= 100; j++) { // 枚举第二个变量，j>i避免重复
            // ✅ 把题目要求的条件，翻译成if判断就行！
            if (i + j == target) {
                printf("%d + %d = %d\n", i, j, target);
            }
        }
    }

    // 如果是枚举单个变量，单层for就够：
    // for (int num = 1; num <= 10000; num++) {
    //     if (num是闰年 / num是回文数 / ...符合题目条件) { 记录答案 }
    // }
    return 0;
}
```
> 背诵点：C组的枚举**不用任何优化**！数据范围最多到1e4，直接把所有可能的情况都试一遍就行，核心就是把题目要求翻译成if判断，剩下的交给循环穷举，送分题。

### 6. 冒泡排序完整模板【官方大纲明确考点：冒泡排序[2]】
```c
#include <stdio.h>
int main() {
    int n;
    scanf("%d", &n);
    int a[10005];
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    // 👇 冒泡排序核心代码【大纲必考，必须背】
    for (int i = 0; i < n - 1; i++) {         // 外层：总共排n-1轮
        for (int j = 0; j < n - 1 - i; j++) { // 内层：每轮把最大的数"冒泡"到最后
            if (a[j] > a[j+1]) {              // 相邻元素比较，前大后小就交换
                int tmp = a[j];
                a[j] = a[j+1];
                a[j+1] = tmp;
            }
        }
    }

    // 输出排序后的结果
    for (int i = 0; i < n; i++) {
        printf("%d ", a[i]);
    }
    printf("\n");
    return 0;
}
```
> 背诵点：冒泡排序的核心是**相邻元素两两比，大的往后放**，每轮把当前最大的数放到末尾，n个数只要排n-1轮，背会这个，C组考排序题直接套，大纲明确要求的考点，考到就是送分。
