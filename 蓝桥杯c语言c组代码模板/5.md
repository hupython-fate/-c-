### 第二梯队5. 日期时间模拟：闰年判断+间隔天数计算
```c
#include <stdio.h>

// 1. 闰年判断，C组日期题必考
int is_leap(int year) {
    // 能被4整除但不能被100，或者能被400整除，就是闰年
    return (year % 4 == 0 && year % 100 != 0) || (year % 400 == 0);
}

// 平年每个月的天数，索引对应月份，1月就是31，2月28...
int month_days[] = {0,31,28,31,30,31,30,31,31,30,31,30,31};

// 2. 计算从公元1年到y-m-d的总天数，用来算间隔
int get_total_days(int y, int m, int d) {
    int days = 0;
    // 加之前所有年的天数
    for (int i = 1; i < y; i++) {
        days += is_leap(i) ? 366 : 365;
    }
    // 加之前所有月的天数
    for (int i = 1; i < m; i++) {
        days += month_days[i];
        if (i == 2 && is_leap(y)) days += 1; // 闰年2月多1天
    }
    // 加当月的天数
    days += d;
    return days;
}

int main() {
    // 示例：输入两个日期，算间隔天数
    int y1,m1,d1, y2,m2,d2;
    scanf("%d%d%d%d%d%d", &y1,&m1,&d1, &y2,&m2,&d2);
    int days1 = get_total_days(y1,m1,d1);
    int days2 = get_total_days(y2,m2,d2);
    printf("两个日期的间隔天数：%d\n", days2 - days1);
    return 0;
}
```
> 背诵点：C组日期题的万能模板！不管是算间隔、算星期几，核心都是把日期转成总天数，相减就行。记住闰年的判断条件，还有月份的天数数组，背会这个，日期题直接套，不用自己推。

### 第二梯队6. 基础数学：累加/累乘/等差数列求和
```c
#include <stdio.h>
int main() {
    int n;
    scanf("%d", &n);

    // 1. 1到n的累加和，直接用公式，不用循环，速算！
    // 公式：n*(n+1)/2，用long long防溢出
    long long sum_1_n = (long long)n * (n + 1) / 2;
    printf("1到%d的累加和：%lld\n", n, sum_1_n);

    // 2. 等差数列求和，通用公式：(首项+末项)*项数/2
    int a1, an, cnt;
    scanf("%d%d%d", &a1, &an, &cnt);
    long long arith_sum = (long long)(a1 + an) * cnt / 2;
    printf("等差数列的和：%lld\n", arith_sum);

    // 3. 1到n的累乘（通用累乘模板）
    long long mul = 1;
    for (int i = 1; i <= n; i++) {
        mul *= i;
    }
    printf("1到%d的累乘：%lld\n", n, mul);

    return 0;
}
```
> 背诵点：这些都是填空速算用的，不用写循环一个个加，直接套公式，1秒出结果，省时间。记得用long long，避免数太大溢出，C组的题经常会有这点坑。

### 第三梯队1. ASCII字符判断（大小写/数字/空格区分）
```c
#include <stdio.h>

// 以下判断直接背，不用记ASCII码，用字符比较就行
int is_upper(char c) { // 判断是不是大写字母 A-Z
    return c >= 'A' && c <= 'Z';
}
int is_lower(char c) { // 判断是不是小写字母 a-z
    return c >= 'a' && c <= 'z';
}
int is_digit(char c) { // 判断是不是数字 0-9
    return c >= '0' && c <= '9';
}
int is_space(char c) { // 判断是不是空格
    return c == ' ';
}

int main() {
    char str[1005];
    gets(str); // 读入带空格的字符串
    int upper_cnt=0, lower_cnt=0, digit_cnt=0, space_cnt=0;
    for (int i=0; str[i]!='\0'; i++) {
        if (is_upper(str[i])) upper_cnt++;
        else if (is_lower(str[i])) lower_cnt++;
        else if (is_digit(str[i])) digit_cnt++;
        else if (is_space(str[i])) space_cnt++;
    }
    printf("大写：%d，小写：%d，数字：%d，空格：%d\n", upper_cnt, lower_cnt, digit_cnt, space_cnt);
    return 0;
}
```
> 背诵点：这个是字符串统计题的辅助，比如让你统计字符串里的大写字母、小写、数字个数，直接用这些判断就行，不用记ASCII的数字，直接用字符比，超级好背，遇到这种题直接套。
