### 补全1. 插入排序完整模板（C组排序第三个考点，补上之前没给全的排序）
```c
#include <stdio.h>
int main() {
    int n;
    scanf("%d", &n);
    int a[10005];
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    // 👇 插入排序核心代码
    for (int i = 1; i < n; i++) { // 从第二个元素开始，前面的已经是排好序的
        int tmp = a[i]; // 拿出当前要插入的元素
        int j = i - 1;
        // 把前面比tmp大的元素，都往后挪一位
        while (j >= 0 && a[j] > tmp) {
            a[j + 1] = a[j];
            j--;
        }
        // 把tmp插入到正确的位置
        a[j + 1] = tmp;
    }

    // 输出排序后的结果
    for (int i = 0; i < n; i++) {
        printf("%d ", a[i]);
    }
    printf("\n");
    return 0;
}
```
> 背诵点：至此C组要求的三个排序就全齐了！插入排序的核心是：**前面的已经排好序了，把当前元素插进去，大的往后挪**，背会这个，排序题就全覆盖了。

### 补全2. DFS深度优先搜索基础模板（C组入门版，迷宫/全排列通用）
```c
#include <stdio.h>
#include <string.h>
int n, m;               // 迷宫的行、列
int maze[105][105];     // 迷宫：0是可走的路，1是墙
int vis[105][105];      // 访问标记，避免走回头路
// 四个方向：上下左右，直接背这个数组
int dx[] = {-1, 1, 0, 0};
int dy[] = {0, 0, -1, 1};

// DFS函数：当前在(x,y)，能不能走到终点(ex,ey)
int dfs(int x, int y, int ex, int ey) {
    if (x == ex && y == ey) return 1; // 走到终点了，返回成功
    vis[x][y] = 1; // 标记当前点已经走过了

    // 遍历四个方向，挨个试
    for (int i = 0; i < 4; i++) {
        int nx = x + dx[i];
        int ny = y + dy[i];
        // 检查：新点在迷宫内、不是墙、没走过
        if (nx >=0 && nx <n && ny >=0 && ny <m && maze[nx][ny]==0 && !vis[nx][ny]) {
            if (dfs(nx, ny, ex, ey)) return 1; // 这个方向能走到终点，直接返回
        }
    }
    return 0; // 所有方向都走不通，返回失败
}

int main() {
    scanf("%d%d", &n, &m);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            scanf("%d", &maze[i][j]);
        }
    }
    int sx, sy, ex, ey; // 起点坐标、终点坐标
    scanf("%d%d%d%d", &sx, &sy, &ex, &ey);
    
    memset(vis, 0, sizeof(vis));
    if (dfs(sx, sy, ex, ey)) {
        printf("可以走到终点！\n");
    } else {
        printf("走不到终点！\n");
    }
    return 0;
}
```
> 背诵点：DFS的核心就是**递归搜所有方向，标记访问过的点**，C组的迷宫、全排列题，直接套这个就行，四个方向的数组直接背，不用自己推，访问标记别忘，不然会走回头路。

### 补全3. BFS广度优先搜索基础模板（C组入门版，最短路径通用）
```c
#include <stdio.h>
#include <string.h>
int n, m;
int maze[105][105];
int vis[105][105];
// 同样的四个方向，和DFS共用，背一次就行
int dx[] = {-1, 1, 0, 0};
int dy[] = {0, 0, -1, 1};

// 队列的节点，存坐标和当前步数
typedef struct {
    int x, y;
    int step;
} Node;
Node q[10005]; // 用数组模拟队列，C组够用，不用搞复杂的库

// BFS函数：找起点到终点的最短步数
int bfs(int sx, int sy, int ex, int ey) {
    int head = 0, tail = 0; // 队列的头、尾指针
    // 起点入队
    q[tail].x = sx;
    q[tail].y = sy;
    q[tail].step = 0;
    tail++;
    vis[sx][sy] = 1;

    while (head < tail) {
        Node cur = q[head]; // 队头出队
        head++;
        if (cur.x == ex && cur.y == ey) return cur.step; // 到终点了，返回步数

        // 遍历四个方向
        for (int i = 0; i < 4; i++) {
            int nx = cur.x + dx[i];
            int ny = cur.y + dy[i];
            if (nx >=0 && nx <n && ny >=0 && ny <m && maze[nx][ny]==0 && !vis[nx][ny]) {
                vis[nx][ny] = 1;
                // 新点入队
                q[tail].x = nx;
                q[tail].y = ny;
                q[tail].step = cur.step + 1;
                tail++;
            }
        }
    }
    return -1; // 走不到终点
}

int main() {
    scanf("%d%d", &n, &m);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            scanf("%d", &maze[i][j]);
        }
    }
    int sx, sy, ex, ey;
    scanf("%d%d%d%d", &sx, &sy, &ex, &ey);
    
    memset(vis, 0, sizeof(vis));
    int step = bfs(sx, sy, ex, ey);
    if (step != -1) {
        printf("最短路径步数：%d\n", step);
    } else {
        printf("走不到终点！\n");
    }
    return 0;
}
```
> 背诵点：BFS的核心是**队列，一层一层搜，第一次到终点就是最短的**，和DFS用同一个方向数组，背一次就行，用数组模拟队列，头出尾进，C组的最短步、层序题，直接套这个。
