### 补全7. 高精度加法基础模板（补上高精度考点，大整数加法）
```c
#include <stdio.h>
#include <string.h>
int main() {
    // 示例：两个超大的整数相加，超过long long的范围，用高精度
    char a[1005], b[1005];
    scanf("%s%s", a, b); // 用字符串读入大整数

    int num1[1005] = {0}, num2[1005] = {0}, res[1005] = {0};
    int len1 = strlen(a), len2 = strlen(b);
    // 👇 核心：把字符串逆序转成数字数组，低位在前，方便进位
    for (int i = 0; i < len1; i++) {
        num1[i] = a[len1 - 1 - i] - '0';
    }
    for (int i = 0; i < len2; i++) {
        num2[i] = b[len2 - 1 - i] - '0';
    }

    // 逐位相加，处理进位
    int max_len = len1 > len2 ? len1 : len2;
    for (int i = 0; i < max_len; i++) {
        res[i] += num1[i] + num2[i];
        if (res[i] >= 10) { // 进位
            res[i+1] += res[i] / 10;
            res[i] %= 10;
        }
    }

    // 逆序输出结果，去掉前导0
    int i = max_len;
    while (i > 0 && res[i] == 0) i--;
    for (; i >= 0; i--) {
        printf("%d", res[i]);
    }
    printf("\n");
    return 0;
}
```
> 背诵点：高精度的核心就是**用数组模拟手工加法**，因为数太大，普通的整数存不下，所以把大整数转成数组，逆序存，然后逐位加、处理进位，C组的高精度一般就考加法，背会这个直接套就行。

### 补全8. 栈基础操作模板（补上栈考点）
```c
#include <stdio.h>
int stack[10005]; // 用数组模拟栈，C组够用
int top = -1;     // 栈顶指针，-1表示空栈

// 入栈：把x放到栈顶
void push(int x) {
    stack[++top] = x;
}
// 出栈：把栈顶的元素拿出来
int pop() {
    return stack[top--];
}
// 取栈顶元素，不删除
int get_top() {
    return stack[top];
}
// 判断栈是不是空
int is_empty() {
    return top == -1;
}

int main() {
    // 示例：栈的基础操作，栈是后进先出
    push(1);
    push(2);
    push(3);
    printf("当前栈顶：%d\n", get_top()); // 输出3，最后进去的最先出来
    pop();
    printf("出栈后栈顶：%d\n", get_top()); // 输出2
    printf("栈是不是空：%d\n", is_empty()); // 输出0，不是空
    return 0;
}
```
> 背诵点：栈的核心是**后进先出**，用数组模拟，top指针，入栈top加，出栈top减，背会这几个函数，C组的栈题都是基础操作，直接套就行，不会考复杂的。

### 补全9. 队列基础操作模板（补上队列考点）
```c
#include <stdio.h>
int q[10005]; // 用数组模拟队列
int head = 0; // 队头指针，出队的位置
int tail = 0; // 队尾指针，入队的位置

// 入队：把x放到队尾
void enqueue(int x) {
    q[tail++] = x;
}
// 出队：把队头的元素拿出来
int dequeue() {
    return q[head++];
}
// 取队头元素，不删除
int get_front() {
    return q[head];
}
// 判断队列是不是空
int is_empty() {
    return head == tail;
}

int main() {
    // 示例：队列的基础操作，队列是先进先出
    enqueue(1);
    enqueue(2);
    enqueue(3);
    printf("当前队头：%d\n", get_front()); // 输出1，最先进去的最先出来
    dequeue();
    printf("出队后队头：%d\n", get_front()); // 输出2
    printf("队列是不是空：%d\n", is_empty()); // 输出0，不是空
    return 0;
}
```
> 背诵点：队列的核心是**先进先出**，用数组模拟，head管出队，tail管入队，背会这几个函数，C组的队列题都是基础操作，直接套就行。
