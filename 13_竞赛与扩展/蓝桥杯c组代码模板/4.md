### 第二梯队2. 基础查找：顺序查找+极简二分查找
```c
#include <stdio.h>

// 1. 顺序查找：无序数组通用，找到返回索引，找不到返回-1
int seq_search(int a[], int n, int target) {
    for (int i = 0; i < n; i++) {
        if (a[i] == target) return i;
    }
    return -1;
}

// 2. 二分查找：必须是升序有序数组！C组够用的极简版
int binary_search(int a[], int n, int target) {
    int left = 0, right = n - 1;
    while (left <= right) {
        int mid = (left + right) / 2; // 取中间位置
        if (a[mid] == target) return mid;
        else if (a[mid] < target) left = mid + 1; // 目标在右半区
        else right = mid - 1; // 目标在左半区
    }
    return -1;
}

int main() {
    int n, target;
    scanf("%d", &n);
    int a[10005];
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
    scanf("%d", &target);

    printf("顺序查找结果(无序用)：%d\n", seq_search(a, n, target));
    printf("二分查找结果(有序用)：%d\n", binary_search(a, n, target));
    return 0;
}
```
> 背诵点：无序数组用顺序查找，直接遍历就行；有序数组用二分，核心是左右指针缩范围，C组的查找题基本就这两种，背会这两个函数，遇到题直接调用。

### 第二梯队3. 斐波那契数列（迭代+递归双版本）
```c
#include <stdio.h>

// 递归版：n比较小的时候用，代码极简单
int fib_recur(int n) {
    if (n == 1 || n == 2) return 1; // 基线：前两项都是1
    return fib_recur(n-1) + fib_recur(n-2);
}

// 迭代版：n大的时候用，不会栈溢出，效率更高
int fib_iter(int n) {
    if (n == 1 || n == 2) return 1;
    int a = 1, b = 1; // 存前两项的值
    int res;
    for (int i = 3; i <= n; i++) {
        res = a + b;
        a = b;
        b = res;
    }
    return res;
}

int main() {
    int n;
    scanf("%d", &n);
    printf("递归版：第%d项斐波那契数 = %d\n", n, fib_recur(n));
    printf("迭代版：第%d项斐波那契数 = %d\n", n, fib_iter(n));
    return 0;
}
```
> 背诵点：斐波那契的核心公式就是`f(n) = f(n-1) + f(n-2)`，前两项是1。C组的兔子繁殖、爬楼梯题，全都是这个模板改的，n小用递归，n大用迭代，直接套。

### 第二梯队4. 阶乘计算（循环+递归双版本）
```c
#include <stdio.h>

// 递归版阶乘，n小的时候用
int fact_recur(int n) {
    if (n == 0 || n == 1) return 1; // 0!和1!都是1，基线条件
    return n * fact_recur(n - 1);
}

// 迭代版阶乘，n大的时候用，不会溢出
int fact_iter(int n) {
    int res = 1;
    for (int i = 2; i <= n; i++) {
        res *= i;
    }
    return res;
}

int main() {
    int n;
    scanf("%d", &n);
    printf("递归版：%d! = %d\n", n, fact_recur(n));
    printf("迭代版：%d! = %d\n", n, fact_iter(n));
    return 0;
}
```
> 背诵点：阶乘的核心就是从1乘到n，记住0的阶乘是1就行。递归的好写，迭代的更安全，C组的填空、排列基础题，直接用这个模板。
