### 补全4. 通用模拟题基础模板（补上模拟考点，过程模拟通用）
```c
#include <stdio.h>
int main() {
    // 示例：约瑟夫环问题（C组最常考的模拟题之一）
    // n个人围成圈，数到m的出队，最后剩下的是谁
    int n, m;
    scanf("%d%d", &n, &m);
    int people[105];
    for (int i = 0; i < n; i++) {
        people[i] = i + 1; // 初始化所有人的编号
    }

    int idx = 0;    // 当前数到的位置
    int remain = n; // 剩下的人数
    // 👇 模拟的核心：题目说什么，你就用代码一步步复现什么
    while (remain > 1) {
        // 数m次，找到要出队的人的位置
        idx = (idx + m - 1) % remain;
        // 模拟出队：后面的人往前挪一位，把出队的删掉
        for (int i = idx; i < remain - 1; i++) {
            people[i] = people[i + 1];
        }
        remain--; // 剩下的人少一个
    }

    printf("最后剩下的人编号：%d\n", people[0]);
    return 0;
}
```
> 背诵点：模拟题没有通用算法，核心就是**题目说什么，你就用代码一步步写什么**，比如排队、游戏回合、约瑟夫环，全都是把题目描述的过程，翻译成代码一步步执行，不用想复杂的优化，C组的模拟题数据都很小，直接模拟就行。

### 补全5. 贪心通用入门模板（补上详细的贪心，比之前的更通用）
```c
#include <stdio.h>
#include <stdlib.h>
// 活动的结构体：开始时间、结束时间
typedef struct {
    int start;
    int end;
} Activity;

// 排序的比较函数：按活动的结束时间从小到大排
int cmp(const void *a, const void *b) {
    Activity *aa = (Activity*)a;
    Activity *bb = (Activity*)b;
    return aa->end - bb->end;
}

int main() {
    // 示例：活动选择问题（贪心最经典的入门题，C组常考）
    // 给你n个活动，每个有开始结束时间，最多能选多少个不冲突的活动
    int n;
    scanf("%d", &n);
    Activity acts[105];
    for (int i = 0; i < n; i++) {
        scanf("%d%d", &acts[i].start, &acts[i].end);
    }

    // 👇 贪心核心：先排序，然后每次选当前最优的，给后面留空间
    qsort(acts, n, sizeof(Activity), cmp); // 按结束时间排序
    int cnt = 0;
    int last_end = -1; // 上一个选的活动的结束时间
    for (int i = 0; i < n; i++) {
        // 选结束最早的、和上一个不冲突的活动
        if (acts[i].start >= last_end) {
            cnt++;
            last_end = acts[i].end;
        }
    }

    printf("最多能选%d个不冲突的活动\n", cnt);
    return 0;
}
```
> 背诵点：通用贪心的核心就是**先把选项排序，然后每次选当前最优的，不影响后面的**，比如活动选择选结束最早的，找零钱选面额最大的，都是这个逻辑，背会这个，C组的贪心题就全覆盖了，比之前的找零钱更通用。

### 补全6. 一维DP基础模板（补上普通一维DP考点）
```c
#include <stdio.h>
#include <limits.h>
int main() {
    // 示例1：爬楼梯问题（最基础的一维DP）
    // n阶台阶，一次走1或2步，有多少种走法
    int n;
    scanf("%d", &n);
    int dp[1005]; // dp[i]表示：走到i阶台阶，有多少种方法
    dp[1] = 1;
    dp[2] = 2;
    for (int i = 3; i <= n; i++) {
        // 👇 一维DP核心：用前面的状态，推当前的状态
        dp[i] = dp[i-1] + dp[i-2]; 
        // 到i阶，要么从i-1走1步，要么从i-2走2步，加起来就是总方法
    }
    printf("爬楼梯的方法数：%d\n", dp[n]);

    // 示例2：最大子段和（另一个经典一维DP）
    int m;
    scanf("%d", &m);
    int a[1005];
    for (int i = 0; i < m; i++) {
        scanf("%d", &a[i]);
    }
    int dp2[1005]; // dp2[i]表示：以i结尾的最大子段和
    dp2[0] = a[0];
    int max_sum = dp2[0];
    for (int i = 1; i < m; i++) {
        dp2[i] = a[i] > (dp2[i-1] + a[i]) ? a[i] : (dp2[i-1] + a[i]);
        if (dp2[i] > max_sum) max_sum = dp2[i];
    }
    printf("最大子段和：%d\n", max_sum);
    return 0;
}
```
> 背诵点：一维DP的核心就是**dp[i]存状态，用前面的状态推当前的**，不用搞复杂的，C组的DP都是这种入门的，比如爬楼梯、最大子段和，背会递推的逻辑，遇到题直接套就行。
