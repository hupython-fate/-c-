### 第三梯队2. 结构体基础简单使用
```c
#include <stdio.h>
#include <string.h>

// 定义结构体，C组的结构体都是这种简单的自定义类型，比如存学生信息
typedef struct {
    int id;         // 学号
    char name[50];  // 姓名
    int score;      // 成绩
} Student; // 别名，之后直接用Student定义变量就行

int main() {
    // 1. 单个结构体的输入输出
    Student s;
    scanf("%d %s %d", &s.id, s.name, &s.score);
    printf("单个学生：学号=%d，姓名=%s，成绩=%d\n", s.id, s.name, s.score);

    // 2. 结构体数组，存多个学生的信息
    Student stu_arr[100];
    int n;
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        scanf("%d %s %d", &stu_arr[i].id, stu_arr[i].name, &stu_arr[i].score);
    }

    return 0;
}
```
> 背诵点：C组的结构体题非常简单，就是定义、输入、输出，不会考复杂的链表、指针，背会这个，遇到结构体题直接套就行，偶尔考，保底用。

### 第三梯队3. 简单贪心思想入门模板
```c
#include <stdio.h>
int main() {
    // 示例：找零钱问题，C组最常见的简单贪心题
    // 硬币面额：50、25、10、5、1，找n块钱，最少用多少个硬币
    int coins[] = {50, 25, 10, 5, 1}; // 先把面额从大到小排序
    int n;
    scanf("%d", &n);

    int cnt = 0;
    // 贪心核心：每次选当前能选的最大的，局部最优推全局最优
    for (int i = 0; i < 5; i++) {
        if (n >= coins[i]) {
            int num = n / coins[i]; // 能拿多少个当前的大硬币
            cnt += num;
            n -= num * coins[i];
        }
        if (n == 0) break;
    }

    printf("最少需要%d个硬币\n", cnt);
    return 0;
}
```
> 背诵点：C组的贪心都是这种入门级的，核心就是**把选项从大到小排序，每次拿最大的**，不用管后面的，比如分糖果、分物资、找零钱，都是这个逻辑，背会这个，遇到简单贪心题直接套。

### 第三梯队4. 一维前缀和简易模板
```c
#include <stdio.h>
int main() {
    int n;
    scanf("%d", &n);
    int a[10005];
    int pre[10005] = {0}; // 前缀和数组，pre[i]表示前i个元素的和

    // 前缀和的递推计算，从1开始索引，方便计算
    for (int i = 1; i <= n; i++) {
        scanf("%d", &a[i]);
        pre[i] = pre[i-1] + a[i];
    }

    // 求l到r的区间和，O(1)直接出结果，不用再遍历一遍
    int l, r;
    scanf("%d%d", &l, &r);
    int sum = pre[r] - pre[l-1];
    printf("区间[%d, %d]的和：%d\n", l, r, sum);

    return 0;
}
```
> 背诵点：前缀和是用来优化区间求和的，比如多次查询区间和，不用每次都遍历数组，提前算好前缀和，直接减就行。核心就两句：`pre[i] = pre[i-1] + a[i]`，区间和`sum = pre[r] - pre[l-1]`，超级好背，偶尔遇到优化题直接用。
